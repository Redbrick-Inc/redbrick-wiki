# Return to first screen at the end of the game

<br />
<center>![](../../../images/example-victory-retry.gif) Example</center>

## Code

```js showLineNumbers copy
const item = WORLD.getObject("item");
const startbtn = GUI.getObject("ballstopcle_button_play_on(e77)");
const success = GUI.getObject("good_d(b37)");
const retrybtn = GUI.getObject("button_text_large_yellow_retry(8c4)");

success.hide();
retrybtn.hide();
startbtn.onClick(() => {
  startbtn.hide();
});

function Start() {
  item.onCollide(PLAYER, () => {
    item.kill();
    success.show();
    setTimeout(() => {
      retrybtn.show();
    }, 2000);
  });

  retrybtn.onClick(() => {
    success.hide();
    PLAYER.spawn();
    item.revive();

    retrybtn.hide();
    startbtn.show();
  });
}
```
