### Code

```js showLineNumbers copy
const falseboxes = [];
for (let i = 0; i < 5; i++) {
  falseboxes[i] = WORLD.getObject("falsebox_" + i);
}

function Start() {
  falseboxes.forEach((box) => {
    box.onCollide(PLAYER, () => {
      box.kill();
      setTimeout(() => {
        box.revive();
      }, 1000);
    });
  });
}
```
